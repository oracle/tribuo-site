048/**
049 * A {@link DataSource} for loading separable data from a text file (e.g. CSV, TSV)
050 * and applying {@link FieldProcessor}s to it.
051 */
052public class CSVDataSource<T extends Output<T>> extends ColumnarDataSource<T> {
053
054    private URI dataFile;
055
056    @Config(mandatory = true,description="Path to the CSV file.")
057    private Path dataPath;
058
059    @Config(description="The CSV separator character.")
060    private char separator = CSVIterator.SEPARATOR;
061
062    @Config(description="The CSV quote character.")
063    private char quote = CSVIterator.QUOTE;
064
065    private ConfiguredDataSourceProvenance provenance;
066
067    /**
068     * For OLCUT.
069     */
070    private CSVDataSource() {}
071
072    /**
073     * Creates a CSVDataSource using the specified RowProcessor to process the data.
074     *
075     * <p>
076     *
077     * Uses ',' as the separator, '"' as the quote character, and '\' as the escape character.
078     * @param dataPath The Path to the data file.
079     * @param rowProcessor The row processor which converts a row into an {@link Example}.
080     * @param outputRequired Is the output required to exist in the data file.
081     */
082    public CSVDataSource(Path dataPath, RowProcessor<T> rowProcessor, boolean outputRequired) {
083        this(dataPath,rowProcessor,outputRequired, CSVIterator.SEPARATOR, CSVIterator.QUOTE);
084    }
085
086    /**
087     * Creates a CSVDataSource using the specified RowProcessor to process the data.
088     *
089     * <p>
090     *
091     * Uses ',' as the separator, '"' as the quote character, and '\' as the escape character.
092     * @param dataFile A URI for the data file.
093     * @param rowProcessor The row processor which converts a row into an {@link Example}.
094     * @param outputRequired Is the output required to exist in the data file.
095     */
096    public CSVDataSource(URI dataFile, RowProcessor<T> rowProcessor, boolean outputRequired) {
097        this(dataFile,rowProcessor,outputRequired, CSVIterator.SEPARATOR, CSVIterator.QUOTE);
098    }
099
100    /**
101     * Creates a CSVDataSource using the specified RowProcessor to process the data.
102     *
103     * <p>
104     *
105     * Uses '"' as the quote character, and '\' as the escape character.
106     * @param dataPath The Path to the data file.
107     * @param rowProcessor The row processor which converts a row into an {@link Example}.
108     * @param outputRequired Is the output required to exist in the data file.
109     * @param separator The separator character in the data file.
110     */
111    public CSVDataSource(Path dataPath, RowProcessor<T> rowProcessor, boolean outputRequired, char separator) {
112        this(dataPath,rowProcessor,outputRequired,separator, CSVIterator.QUOTE);
113    }
114
115    /**
116     * Creates a CSVDataSource using the specified RowProcessor to process the data.
117     *
118     * <p>

120     * Uses '"' as the quote character, and '\' as the escape character.

122     * @param rowProcessor The row processor which converts a row into an {@link Example}.

124     * @param separator The separator character in the data file.
125     */
126    public CSVDataSource(URI dataFile, RowProcessor<T> rowProcessor, boolean outputRequired, char separator) {
127        this(dataFile,rowProcessor,outputRequired,separator, CSVIterator.QUOTE);
128    }
129
130    /**
131     * Creates a CSVDataSource using the specified RowProcessor to process the data, and the supplied separator and quote
132     * characters to read the input data file.
133     * @param dataFile A URI for the data file.
134     * @param rowProcessor The row processor which converts a row into an {@link Example}.
135     * @param outputRequired Is the output required to exist in the data file.
136     * @param separator The separator character in the data file.
137     * @param quote The quote character in the data file.
138     */
139    public CSVDataSource(URI dataFile, RowProcessor<T> rowProcessor, boolean outputRequired, char separator, char quote) {
140        this(dataFile, Paths.get(dataFile),rowProcessor,outputRequired,separator,quote);
141    }
142
143    /**
144     * Creates a CSVDataSource using the specified RowProcessor to process the data, and the supplied separator and quote
145     * characters to read the input data file.
146     * @param dataPath The Path to the data file.
147     * @param rowProcessor The row processor which converts a row into an {@link Example}.
148     * @param outputRequired Is the output required to exist in the data file.
149     * @param separator The separator character in the data file.
150     * @param quote The quote character in the data file.
151     */
152    public CSVDataSource(Path dataPath, RowProcessor<T> rowProcessor, boolean outputRequired, char separator, char quote) {
153        this(dataPath.toUri(),dataPath,rowProcessor,outputRequired,separator,quote);
154    }
155
156    /**
157     * Creates a CSVDataSource using the specified RowProcessor to process the data, and the supplied separator, quote
158     * characters to read the input data file.
159     * @param dataFile A URI for the data file.
160     * @param rowProcessor The row processor which converts a row into an {@link Example}.
161     * @param outputRequired Is the output required to exist in the data file.
162     * @param separator The separator character in the data file.
163     * @param quote The quote character in the data file.
164     */
165    private CSVDataSource(URI dataFile, Path dataPath, RowProcessor<T> rowProcessor, boolean outputRequired, char separator, char quote) {
166        super(rowProcessor.getResponseProcessor().getOutputFactory(), rowProcessor, outputRequired);
167        this.dataPath = dataPath;
168        this.dataFile = dataFile;
169        this.separator = separator;
170        this.quote = quote;
171        this.provenance = new CSVDataSourceProvenance(this);
172    }
173
174    @Override
175    public void postConfig() {
176        this.dataFile = dataPath.toUri();
177        this.provenance = new CSVDataSourceProvenance(this);
178    }
179
180    @Override
181    public String toString() {
182        return "CSVDataSource(file=" + dataFile + ",rowProcessor="+rowProcessor.getDescription()+")";
183    }
184
185    @Override
186    public ColumnarIterator rowIterator() {
187        try {
188            return new CSVIterator(dataFile, separator, quote);
189        } catch (IOException e) {
190            throw new IllegalStateException("Failed to read data",e);
191        }
192    }
193
194    @Override
195    public ConfiguredDataSourceProvenance getProvenance() {
196        return provenance;
197    }
198
199    public static class CSVDataSourceProvenance extends SkeletalConfiguredObjectProvenance implements ConfiguredDataSourceProvenance {
200        private static final long serialVersionUID = 1L;
201
202        private final DateTimeProvenance fileModifiedTime;
203        private final DateTimeProvenance dataSourceCreationTime;
204        private final HashProvenance sha256Hash;
205
206        <T extends Output<T>> CSVDataSourceProvenance(CSVDataSource<T> host) {
207            super(host,"DataSource");
208            this.fileModifiedTime = new DateTimeProvenance(FILE_MODIFIED_TIME,OffsetDateTime.ofInstant(Instant.ofEpochMilli(host.dataPath.toFile().lastModified()), ZoneId.systemDefault()));
209            this.dataSourceCreationTime = new DateTimeProvenance(DATASOURCE_CREATION_TIME,OffsetDateTime.now());
210            this.sha256Hash = new HashProvenance(DEFAULT_HASH_TYPE,RESOURCE_HASH,ProvenanceUtil.hashResource(DEFAULT_HASH_TYPE,host.dataPath));
211        }
212
213        public CSVDataSourceProvenance(Map<String,Provenance> map) {
214            this(extractProvenanceInfo(map));
215        }
216
217        private CSVDataSourceProvenance(ExtractedInfo info) {
218            super(info);
219            this.fileModifiedTime = (DateTimeProvenance) info.instanceValues.get(FILE_MODIFIED_TIME);
220            this.dataSourceCreationTime = (DateTimeProvenance) info.instanceValues.get(DATASOURCE_CREATION_TIME);
221            this.sha256Hash = (HashProvenance) info.instanceValues.get(RESOURCE_HASH);
222        }
223
224        protected static ExtractedInfo extractProvenanceInfo(Map<String,Provenance> map) {
225            Map<String,Provenance> configuredParameters = new HashMap<>(map);
226            String className = ObjectProvenance.checkAndExtractProvenance(configuredParameters,CLASS_NAME, StringProvenance.class, CSVDataSourceProvenance.class.getSimpleName()).getValue();
227            String hostTypeStringName = ObjectProvenance.checkAndExtractProvenance(configuredParameters,HOST_SHORT_NAME, StringProvenance.class, CSVDataSourceProvenance.class.getSimpleName()).getValue();
228
229            Map<String,PrimitiveProvenance<?>> instanceParameters = new HashMap<>();
230            instanceParameters.put(FILE_MODIFIED_TIME,ObjectProvenance.checkAndExtractProvenance(configuredParameters,FILE_MODIFIED_TIME,DateTimeProvenance.class, CSVDataSourceProvenance.class.getSimpleName()));
231            instanceParameters.put(DATASOURCE_CREATION_TIME,ObjectProvenance.checkAndExtractProvenance(configuredParameters,DATASOURCE_CREATION_TIME,DateTimeProvenance.class, CSVDataSourceProvenance.class.getSimpleName()));
232            instanceParameters.put(RESOURCE_HASH,ObjectProvenance.checkAndExtractProvenance(configuredParameters,RESOURCE_HASH,HashProvenance.class, CSVDataSourceProvenance.class.getSimpleName()));
233
234            return new ExtractedInfo(className,hostTypeStringName,configuredParameters,instanceParameters);
235        }
236
237        @Override
238        public boolean equals(Object o) {
239            if (this == o) return true;
240            if (o == null || getClass() != o.getClass()) return false;
241            if (!super.equals(o)) return false;
242            CSVDataSourceProvenance pairs = (CSVDataSourceProvenance) o;
243            return fileModifiedTime.equals(pairs.fileModifiedTime) &&
244                    dataSourceCreationTime.equals(pairs.dataSourceCreationTime) &&
245                    sha256Hash.equals(pairs.sha256Hash);
246        }
247
248        @Override
249        public int hashCode() {
250            return Objects.hash(super.hashCode(), fileModifiedTime, dataSourceCreationTime, sha256Hash);
251        }
252
253        @Override
254        public Map<String, PrimitiveProvenance<?>> getInstanceValues() {
255            Map<String,PrimitiveProvenance<?>> map = super.getInstanceValues();
256
257            map.put(FILE_MODIFIED_TIME,fileModifiedTime);
258            map.put(DATASOURCE_CREATION_TIME,dataSourceCreationTime);
259            map.put(RESOURCE_HASH,sha256Hash);
260
261            return map;
262        }
263    }
